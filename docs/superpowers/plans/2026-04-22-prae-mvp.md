**Goal:** Build the core Prae processing pipeline — HTML denoise + semantic extraction + JSON/Markdown output — with multi-strategy execution, confidence scoring, and basic retry/escalation.

**Architecture:** Plugin-based input source registry with a strategy pipeline (Denoise → Semantic → Output). Confidence scoring drives retry/escalation. Cloud experience hub is deferred; local experience storage only.

- [ ] **Step 2: Write src/input/HTMLInputSource.ts**

```typescript
import { JSDOM } from 'jsdom';
import { v4 as uuidv4 } from 'uuid';
import { InputSource } from './InputSource';
import { ContentItem, ContentHint } from '../types';

export class HTMLInputSource implements InputSource {
  name = 'html-input';
  version = '1.0.0';
  supportedTypes = ['text/html', 'application/html'];

  parse(raw: Uint8Array): ContentItem {
    const decoder = new TextDecoder('utf-8');
    const html = decoder.decode(raw);
    const dom = new JSDOM(html);
    const document = dom.window.document;

    return {
      id: uuidv4(),
      source: this.name,
      raw,
      meta: {
        title: document.title,
        url: document.URL,
        textContent: document.body?.textContent?.trim() ?? '',
      },
      hints: this.detect(raw),
    };
  }

  detect(partial: Uint8Array): ContentHint {
    const decoder = new TextDecoder('utf-8');
    const text = decoder.decode(partial.slice(0, 1024));

    const mimeMatch = text.match(/<!\s*DOCTYPE\s+html/i) || text.match(/<html/i);
    const confidence = mimeMatch ? 0.95 : 0.3;

    return {
      mimeType: mimeMatch ? 'text/html' : undefined,
      possibleTypes: mimeMatch ? ['text/html'] : [],
      confidence,
    };
  }

  validate(item: ContentItem): boolean {
    return item.meta.textContent !== undefined && item.meta.textContent.length > 0;
  }
}
```

import { InputSource } from './InputSource';

export class InputRegistry {
  private sources = new Map<string, InputSource>();

  register(source: InputSource): void {
    this.sources.set(source.name, source);
  }

  get(name: string): InputSource | undefined {
    return this.sources.get(name);
  }

  detectMimeType(raw: Uint8Array): { source: InputSource; confidence: number } | null {
    let bestMatch: { source: InputSource; confidence: number } | null = null;

    for (const source of this.sources.values()) {
      const hint = source.detect(raw);
      if (hint.confidence !== undefined && hint.confidence > (bestMatch?.confidence ?? 0)) {
        bestMatch = { source, confidence: hint.confidence };
      }
    }

    return bestMatch;
  }
}

export class StrategyRegistry {
  private strategies = new Map<string, Strategy>();

  register(strategy: Strategy): void {
    this.strategies.set(strategy.id, strategy);
  }

  get(id: string): Strategy | undefined {
    return this.strategies.get(id);
  }

  getByType(type: StrategyType): Strategy[] {
    return Array.from(this.strategies.values()).filter((s) => s.type === type);
  }

  getAll(): Strategy[] {
    return Array.from(this.strategies.values());
  }

  listByPriority(type?: StrategyType): Strategy[] {
    const strategies = type ? this.getByType(type) : this.getAll();
    return strategies.sort((a, b) => b.config.priority - a.config.priority);
  }
}

- [ ] **Step 3: Write src/strategies/base/StrategyExecutor.ts**

```typescript
import { Strategy, StrategyType } from './Strategy';
import { ContentItem, StrategyExecution } from '../../types';
import { v4 as uuidv4 } from 'uuid';

export interface FusionResult {
  output: unknown;
  agreement: number;
}

export class StrategyExecutor {
  async executeStrategies(
    strategies: Strategy[],
    contentItem: ContentItem
  ): Promise<{ executions: StrategyExecution[]; fusedOutput: unknown }> {
    const executions: StrategyExecution[] = [];
    const results: unknown[] = [];

    for (const strategy of strategies) {
      const execution: StrategyExecution = {
        id: uuidv4(),
        strategyId: strategy.id,
        startedAt: Date.now(),
        success: false,
      };

      try {
        const output = await strategy.execute(contentItem);
        execution.completedAt = Date.now();
        execution.success = true;
        execution.output = output;
        results.push(output);
      } catch (error) {
        execution.completedAt = Date.now();
        execution.error = error instanceof Error ? error.message : 'Unknown error';
      }

      executions.push(execution);
    }

    const fusedOutput = this.fuseResults(results);

    return { executions, fusedOutput };
  }

  private fuseResults(results: unknown[]): unknown {
    const validResults = results.filter((r) => r !== null && r !== undefined);

    if (validResults.length === 0) return null;
    if (validResults.length === 1) return validResults[0];

    return {
      type: 'fused',
      sources: validResults.length,
      data: validResults,
    };
  }
}
```

import { JSDOM } from 'jsdom';
import { Strategy, StrategyType } from '../base/Strategy';
import { ContentItem } from '../../types';

export class HTMLCleanStrategy implements Strategy {
  id = 'html-clean';
  name = 'HTML Clean Strategy';
  type = StrategyType.DENOISE;
  version = '1.0.0';
  config = { enabled: true, priority: 100 };

  canApply(item: unknown): boolean {
    if (!this.isContentItem(item)) return false;
    return item.meta?.textContent !== undefined;
  }

  async execute(item: unknown): Promise<{ cleanedText: string; removedTags: number }> {
    const contentItem = item as ContentItem;
    const decoder = new TextDecoder('utf-8');
    const html = decoder.decode(contentItem.raw);
    const dom = new JSDOM(html);
    const document = dom.window.document;

    const body = document.body;
    const originalLength = body.innerHTML.length;

    const scripts = body.querySelectorAll('script, style, noscript, iframe, object, embed');
    scripts.forEach((el) => el.remove());

    const navElements = body.querySelectorAll('nav, header, footer, aside, .nav, .navigation, .menu');
    navElements.forEach((el) => el.remove());

    const ads = body.querySelectorAll('[class*="ad-"], [id*="ad-"], .advertisement, .ads');
    ads.forEach((el) => el.remove());

    const cleanedText = body.textContent?.trim() ?? '';
    const removedTags = originalLength - body.innerHTML.length;

    return { cleanedText, removedTags };
  }

  private isContentItem(item: unknown): item is ContentItem {
    return typeof item === 'object' && item !== null && 'meta' in item;
  }
}

import { JSDOM } from 'jsdom';
import { Strategy, StrategyType } from '../base/Strategy';
import { ContentItem } from '../../types';

export class NavigationFilterStrategy implements Strategy {
  id = 'nav-filter';
  name = 'Navigation Filter Strategy';
  type = StrategyType.DENOISE;
  version = '1.0.0';
  config = { enabled: true, priority: 90 };

  canApply(item: unknown): boolean {
    if (!this.isContentItem(item)) return false;
    return item.meta?.textContent !== undefined;
  }

  async execute(item: unknown): Promise<{ navRemoved: number; textLength: number }> {
    const contentItem = item as ContentItem;
    const decoder = new TextDecoder('utf-8');
    const html = decoder.decode(contentItem.raw);
    const dom = new JSDOM(html);
    const document = dom.window.document;

    const navSelectors = [
      'nav', 'header', 'footer', 'aside',
      '[role="navigation"]', '[role="banner"]', '[role="contentinfo"]',
      '.nav', '.navigation', '.menu', '.sidebar', '.sidebar-nav',
    ];

    let navRemoved = 0;
    for (const selector of navSelectors) {
      const elements = document.querySelectorAll(selector);
      navRemoved += elements.length;
      elements.forEach((el) => el.remove());
    }

    const textContent = document.body?.textContent?.trim() ?? '';

    return { navRemoved, textLength: textContent.length };
  }

  private isContentItem(item: unknown): item is ContentItem {
    return typeof item === 'object' && item !== null && 'meta' in item;
  }
}

import { ContentItem } from '../../types';

export interface ChunkingConfig {
  targetChunkSize: number;
  overlap: number;
  splitOn: 'sentence' | 'paragraph' | 'token';
}

export interface Chunk {
  id: string;
  index: number;
  text: string;
  startChar: number;
  endChar: number;
  tokenEstimate: number;
}

export class ChunkingStrategy implements Strategy {
  id = 'chunking';
  name = 'Chunking Strategy';
  type = StrategyType.SEMANTIC;
  version = '1.0.0';
  config: { enabled: boolean; priority: number; params?: Record<string, unknown> } = {
    enabled: true,
    priority: 50,
    params: { targetChunkSize: 512, overlap: 64, splitOn: 'sentence' },
  };

  canApply(item: unknown): boolean {
    if (!this.isContentItem(item)) return false;
    return typeof item.meta?.textContent === 'string' && item.meta.textContent.length > 100;
  }

  async execute(item: unknown): Promise<{ chunks: Chunk[]; totalChunks: number }> {
    const contentItem = item as ContentItem;
    const text: string = contentItem.meta?.cleanedText as string || contentItem.meta?.textContent as string;
    const params = this.config.params as ChunkingConfig || { targetChunkSize: 512, overlap: 64, splitOn: 'sentence' };

    const chunks: Chunk[] = [];
    const sentences = this.splitIntoSentences(text);
    let currentChunk = '';
    let currentStart = 0;
    let chunkIndex = 0;

    for (const sentence of sentences) {
      if (currentChunk.length + sentence.length > params.targetChunkSize && currentChunk.length > 0) {
        chunks.push(this.createChunk(chunkIndex++, currentChunk, currentStart, currentStart + currentChunk.length));
        currentStart += currentChunk.length - params.overlap;
        currentChunk = currentChunk.slice(-params.overlap) + sentence;
      } else {
        currentChunk += sentence;
      }
    }

    if (currentChunk.length > 0) {
      chunks.push(this.createChunk(chunkIndex, currentChunk, currentStart, currentStart + currentChunk.length));
    }

    return { chunks, totalChunks: chunks.length };
  }

  private splitIntoSentences(text: string): string[] {
    return text.split(/(?<=[.!?])\s+/).filter((s) => s.length > 0);
  }

  private createChunk(index: number, text: string, start: number, end: number): Chunk {
    return {
      id: `chunk-${index}`,
      index,
      text,
      startChar: start,
      endChar: end,
      tokenEstimate: Math.ceil(text.length / 4),
    };
  }

  private isContentItem(item: unknown): item is ContentItem {
    return typeof item === 'object' && item !== null && 'meta' in item;
  }
}

import { Strategy, StrategyType } from '../base/Strategy';
import { ContentItem } from '../../types';

export class RelevanceFilterStrategy implements Strategy {
  id = 'relevance-filter';
  name = 'Relevance Filter Strategy';
  type = StrategyType.SEMANTIC;
  version = '1.0.0';
  config = { enabled: true, priority: 40 };

  private readonly minLength = 50;
  private readonly maxDuplicateRatio = 0.7;

  canApply(item: unknown): boolean {
    if (!this.isContentItem(item)) return false;
    return typeof item.meta?.textContent === 'string';
  }

  async execute(item: unknown): Promise<{ filteredText: string; removedRatio: number }> {
    const contentItem = item as ContentItem;
    const text: string = contentItem.meta?.cleanedText as string || contentItem.meta?.textContent as string;
    const originalLength = text.length;

    const paragraphs = text.split(/\n\n+/);
    const filtered = paragraphs.filter((p) => this.isRelevant(p));

    const deduplicated = this.deduplicate(filtered);
    const filteredText = deduplicated.join('\n\n');

    const removedRatio = 1 - filteredText.length / originalLength;

    return { filteredText, removedRatio: Math.max(0, removedRatio) };
  }

  private isRelevant(text: string): boolean {
    if (text.length < this.minLength) return false;
    const hasMeaningfulContent = /[a-zA-Z]{3,}/.test(text);
    return hasMeaningfulContent;
  }

  private deduplicate(paragraphs: string[]): string[] {
    const seen = new Set<string>();
    return paragraphs.filter((p) => {
      const normalized = p.toLowerCase().trim();
      if (seen.has(normalized)) return false;
      if (this.isHighlyDuplicated(normalized, seen)) return false;
      seen.add(normalized);
      return true;
    });
  }

  private isHighlyDuplicated(text: string, seen: Set<string>): boolean {
    for (const existing of seen) {
      if (this.calculateSimilarity(text, existing) > this.maxDuplicateRatio) {
        return true;
      }
    }
    return false;
  }

  private calculateSimilarity(a: string, b: string): number {
    if (a === b) return 1;
    if (a.length === 0 || b.length === 0) return 0;

    const aWords = new Set(a.split(/\s+/));
    const bWords = new Set(b.split(/\s+/));
    const intersection = new Set([...aWords].filter((x) => bWords.has(x)));
    const union = new Set([...aWords, ...bWords]);

    return intersection.size / union.size;
  }

  private isContentItem(item: unknown): item is ContentItem {
    return typeof item === 'object' && item !== null && 'meta' in item;
  }
}

import { ConfidenceScore, ConfidenceConfig, StrategyExecution } from '../types';

export class ConfidenceScorer {
  private config: ConfidenceConfig;

  constructor(config?: Partial<ConfidenceConfig>) {
    this.config = {
      thresholds: config?.thresholds ?? { pass: 0.85, retry: 0.6, escalate: 0.4 },
      components: config?.components ?? {
        textQuality: { weight: 0.3, indicators: ['cleanedLength', 'noiseRemoved'] },
        entityExtraction: { weight: 0.25, indicators: ['entitiesFound', 'entityDensity'] },
        structuralIntegrity: { weight: 0.25, indicators: ['structurePreserved', 'paragraphCount'] },
        contextualCoherence: { weight: 0.2, indicators: ['coherence', 'topicContinuity'] },
      },
      historicalWeight: config?.historicalWeight ?? 0.1,
      consistencyBonus: config?.consistencyBonus ?? 0.05,
    };
  }

  calculateScore(
    contentItem: unknown,
    executions: StrategyExecution[],
    metadata?: { historicalScore?: number; strategyAgreement?: number }
  ): ConfidenceScore {
    const components = this.calculateComponents(contentItem, executions);
    const overall = this.weightedSum(components);

    const bonus = this.calculateBonus(metadata?.historicalScore, metadata?.strategyAgreement);
    const finalScore = Math.min(1, overall + bonus);

    return {
      overall: Math.round(finalScore * 100) / 100,
      components,
      bonus,
      isPassing: finalScore >= this.config.thresholds.pass,
    };
  }

  private calculateComponents(
    contentItem: unknown,
    executions: StrategyExecution[]
  ): ConfidenceScore['components'] {
    const meta = (contentItem as any)?.meta || {};

    const textQuality = this.calculateTextQuality(meta, executions);
    const entityExtraction = this.calculateEntityExtraction(meta);
    const structuralIntegrity = this.calculateStructuralIntegrity(meta);
    const contextualCoherence = this.calculateContextualCoherence(meta);

    return { textQuality, entityExtraction, structuralIntegrity, contextualCoherence };
  }

  private calculateTextQuality(meta: Record<string, unknown>, executions: StrategyExecution[]): number {
    const textLength = (meta.cleanedText || meta.textContent || '') as string;
    if (textLength === 0) return 0;

    const successfulStrategies = executions.filter((e) => e.success).length;
    const strategySuccessRate = executions.length > 0 ? successfulStrategies / executions.length : 0;

    const lengthScore = Math.min(1, textLength.length / 1000);
    const successScore = strategySuccessRate * 0.5 + 0.5;

    return Math.round((lengthScore * 0.4 + successScore * 0.6) * 100) / 100;
  }

  private calculateEntityExtraction(meta: Record<string, unknown>): number {
    const text = (meta.cleanedText || meta.textContent || '') as string;
    if (text.length === 0) return 0;

    const words = text.split(/\s+/).filter(Boolean);
    const capitalizedWords = words.filter((w: string) => /^[A-Z]/.test(w));
    const entityRatio = capitalizedWords.length / words.length;

    return Math.round(Math.min(1, entityRatio * 5) * 100) / 100;
  }

  private calculateStructuralIntegrity(meta: Record<string, unknown>): number {
    const text = (meta.cleanedText || meta.textContent || '') as string;
    if (!text) return 0;

    const paragraphs = text.split(/\n\n+/).filter((p) => p.trim().length > 0);
    const hasStructure = paragraphs.length > 1 ? 1 : 0.5;
    const paragraphScore = Math.min(1, paragraphs.length / 5);

    return Math.round((paragraphScore * 0.6 + hasStructure * 0.4) * 100) / 100;
  }

  private calculateContextualCoherence(_meta: Record<string, unknown>): number {
    return 0.75;
  }

  private calculateBonus(historicalScore?: number, strategyAgreement?: number): ConfidenceScore['bonus'] {
    const historicalConsistency = historicalScore
      ? this.config.consistencyBonus * Math.min(1, historicalScore)
      : 0;

    const multiStrategyAgreement = strategyAgreement !== undefined
      ? this.config.consistencyBonus * strategyAgreement
      : 0;

    return { historicalConsistency, multiStrategyAgreement };
  }

  private weightedSum(components: ConfidenceScore['components']): number {
    const { textQuality, entityExtraction, structuralIntegrity, contextualCoherence } = components;
    const { textQuality: w1, entityExtraction: w2, structuralIntegrity: w3, contextualCoherence: w4 } =
      this.config.components;

    const sum = (
      textQuality * w1.weight +
      entityExtraction * w2.weight +
      structuralIntegrity * w3.weight +
      contextualCoherence * w4.weight
    );

    return Math.round(sum * 100) / 100;
  }

  shouldRetry(score: ConfidenceScore): boolean {
    return score.overall < this.config.thresholds.pass && score.overall >= this.config.thresholds.retry;
  }

  shouldEscalate(score: ConfidenceScore): boolean {
    return score.overall < this.config.thresholds.escalate;
  }
}

export interface PipelineConfig {
  maxRetries: number;
  enableCloudEscalation: boolean;
}

export class Pipeline {
  private executor: StrategyExecutor;
  private registry: StrategyRegistry;
  private scorer: ConfidenceScorer;
  private experienceStore: ExperienceStore;
  private config: PipelineConfig;

  constructor(config: Partial<PipelineConfig> = {}) {
    this.executor = new StrategyExecutor();
    this.registry = new StrategyRegistry();
    this.scorer = new ConfidenceScorer();
    this.experienceStore = new ExperienceStore();
    this.config = {
      maxRetries: config.maxRetries ?? 3,
      enableCloudEscalation: config.enableCloudEscalation ?? false,
    };
  }

  registerStrategy(strategy: Strategy): void {
    this.registry.register(strategy);
  }

  async process(contentItem: ContentItem): Promise<ProcessingResult> {
    const startTime = Date.now();
    let retryCount = 0;
    let lastError: string | undefined;

    while (retryCount <= this.config.maxRetries) {
      try {
        const result = await this.executePipeline(contentItem, retryCount);

        if (result.confidence.isPassing || retryCount >= this.config.maxRetries) {
          result.processingTimeMs = Date.now() - startTime;
          result.retryCount = retryCount;

          await this.experienceStore.record(result);

          return result;
        }
      } catch (error) {
        lastError = error instanceof Error ? error.message : 'Unknown error';
      }

      retryCount++;
    }

    return this.createFailedResult(contentItem, startTime, lastError);
  }

  private async executePipeline(
    contentItem: ContentItem,
    retryCount: number
  ): Promise<ProcessingResult> {
    const denoiseStrategies = this.registry.listByPriority(StrategyType.DENOISE);
    const semanticStrategies = this.registry.listByPriority(StrategyType.SEMANTIC);
    const outputStrategies = this.registry.listByPriority(StrategyType.OUTPUT);

    let processedItem = contentItem;
    let allExecutions: ProcessingResult['strategiesUsed'] = [];

    if (denoiseStrategies.length > 0) {
      const { executions, fusedOutput } = await this.executor.executeStrategies(
        denoiseStrategies,
        processedItem
      );
      allExecutions.push(...executions);
      processedItem = this.mergeOutput(processedItem, fusedOutput);
    }

    if (semanticStrategies.length > 0) {
      const { executions, fusedOutput } = await this.executor.executeStrategies(
        semanticStrategies,
        processedItem
      );
      allExecutions.push(...executions);
      processedItem = this.mergeOutput(processedItem, fusedOutput);
    }

    let finalOutput: unknown;
    if (outputStrategies.length > 0) {
      const { executions, fusedOutput } = await this.executor.executeStrategies(
        outputStrategies,
        processedItem
      );
      allExecutions.push(...executions);
      finalOutput = fusedOutput;
    } else {
      finalOutput = processedItem.meta?.cleanedText || processedItem.meta?.textContent;
    }

    const confidence = this.scorer.calculateScore(
      processedItem,
      allExecutions,
      await this.getHistoricalContext(contentItem)
    );

    return {
      id: uuidv4(),
      contentItem: processedItem,
      strategiesUsed: allExecutions,
      fusedOutput: finalOutput,
      confidence,
      outcome: confidence.isPassing ? 'SUCCESS' : 'RETRY_SUCCESS',
      processingTimeMs: 0,
      retryCount,
    };
  }

  private mergeOutput(contentItem: ContentItem, output: unknown): ContentItem {
    if (!output || typeof output !== 'object') return contentItem;

    const outputObj = output as Record<string, unknown>;

    if ('cleanedText' in outputObj) {
      return {
        ...contentItem,
        meta: { ...contentItem.meta, cleanedText: outputObj.cleanedText },
      };
    }

    return contentItem;
  }

  private async getHistoricalContext(
    contentItem: ContentItem
  ): Promise<{ historicalScore?: number; strategyAgreement?: number } | undefined> {
    const record = await this.experienceStore.getLatest(contentItem.source);
    if (!record) return undefined;

    return {
      historicalScore: record.processing.finalConfidence,
      strategyAgreement: record.processing.strategiesUsed.filter((s) => s.success).length /
        Math.max(1, record.processing.strategiesUsed.length),
    };
  }

  private createFailedResult(
    contentItem: ContentItem,
    startTime: number,
    error?: string
  ): ProcessingResult {
    return {
      id: uuidv4(),
      contentItem,
      strategiesUsed: [],
      fusedOutput: null,
      confidence: {
        overall: 0,
        components: {
          textQuality: 0,
          entityExtraction: 0,
          structuralIntegrity: 0,
          contextualCoherence: 0,
        },
        bonus: { historicalConsistency: 0, multiStrategyAgreement: 0 },
        isPassing: false,
      },
      outcome: 'FAILED',
      processingTimeMs: Date.now() - startTime,
      retryCount: this.config.maxRetries,
    };
  }
}

import { createHash } from 'crypto';
import { ExperienceRecord } from './ExperienceRecord';
import { ProcessingResult } from '../types';

export class ExperienceStore {
  private records: Map<string, ExperienceRecord[]> = new Map();

  async record(result: ProcessingResult, tenantId: string = 'default'): Promise<ExperienceRecord> {
    const record: ExperienceRecord = {
      id: result.id,
      tenantId,
      input: {
        sourceType: result.contentItem.source,
        contentType: result.contentItem.hints.mimeType || 'unknown',
        rawHash: this.hashContent(result.contentItem.raw),
        size: result.contentItem.raw.length,
      },
      processing: {
        strategiesUsed: result.strategiesUsed,
        fusionMethod: 'weighted',
        finalConfidence: result.confidence.overall,
        processingTimeMs: result.processingTimeMs,
        retryCount: result.retryCount,
      },
      outcome: result.outcome,
      learning: {
        isLearned: false,
      },
      createdAt: Date.now(),
      updatedAt: Date.now(),
    };

    const key = `${tenantId}:${result.contentItem.source}`;
    const existing = this.records.get(key) || [];
    existing.push(record);
    this.records.set(key, existing);

    return record;
  }

  async getLatest(sourceType: string, tenantId: string = 'default'): Promise<ExperienceRecord | undefined> {
    const key = `${tenantId}:${sourceType}`;
    const records = this.records.get(key);
    if (!records || records.length === 0) return undefined;
    return records[records.length - 1];
  }

  async getByOutcome(
    outcome: ProcessingResult['outcome'],
    tenantId: string = 'default'
  ): Promise<ExperienceRecord[]> {
    const allRecords: ExperienceRecord[] = [];
    for (const [key, records] of this.records.entries()) {
      if (key.startsWith(`${tenantId}:`)) {
        allRecords.push(...records);
      }
    }
    return allRecords.filter((r) => r.outcome === outcome);
  }

  async addHumanFeedback(
    recordId: string,
    feedback: string,
    correctedResult?: ProcessingResult['fusedOutput'],
    tenantId: string = 'default'
  ): Promise<void> {
    for (const records of this.records.values()) {
      const record = records.find((r) => r.id === recordId);
      if (record) {
        record.humanFeedback = { feedback, correctedResult };
        record.learning.isLearned = true;
        record.updatedAt = Date.now();
        return;
      }
    }
  }

  async getLearnableRecords(tenantId: string = 'default'): Promise<ExperienceRecord[]> {
    const allRecords: ExperienceRecord[] = [];
    for (const [key, records] of this.records.entries()) {
      if (key.startsWith(`${tenantId}:`)) {
        allRecords.push(...records);
      }
    }
    return allRecords.filter((r) => r.humanFeedback !== undefined && !r.learning.isLearned);
  }

  private hashContent(content: Uint8Array): string {
    return createHash('sha256').update(Buffer.from(content)).digest('hex').slice(0, 16);
  }
}

async function makeRequest(
  app: any,
  method: string,
  path: string,
  body: any = {},
  apiKey?: string
): Promise<{ statusCode: number; body: any }> {
  return new Promise((resolve) => {
    const http = require('http');
    const server = app.listen(0);
    const { port } = server.address() as any;

    const options = {
      hostname: 'localhost',
      port,
      path,
      method,
      headers: {
        'Content-Type': 'application/json',
        ...(apiKey ? { 'X-API-Key': apiKey } : {}),
      },
    };

    const req = http.request(options, (res: any) => {
      let data = '';
      res.on('data', (chunk: any) => (data += chunk));
      res.on('end', () => {
        server.close();
        try {
          resolve({ statusCode: res.statusCode, body: JSON.parse(data) });
        } catch {
          resolve({ statusCode: res.statusCode, body: data });
        }
      });
    });

    if (Object.keys(body).length > 0) {
      req.write(JSON.stringify(body));
    }
    req.end();
  });
}
